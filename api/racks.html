headers: {

return data.data;
} else {
throw new Error(data.message);
}
}

async function findRackId(rackCode, baseId = null) {

action: 'get_rack_id',
rack_code: rackCode
});

method: 'GET',
headers: {
'Authorization': 'Bearer ' + token
}
});

if (data.code === 200) {
if (data.data.match_type === 'exact') {
return data.data.rack_id;
} else {
return data.data.matches;
}
} else {
throw new Error(data.message);
}
}
